# Convert to datetime first
bikes['date'] = pd.to_datetime(bikes['date'])

# extract components
bikes["year"] = bikes["date"].dt.year
bikes["month"] = bikes["date"].dt.month
bikes["month_name"] = bikes["date"].dt.month_name()   # month label
bikes["day"] = bikes["date"].dt.day
bikes["day_of_week"] = bikes["date"].dt.day_name()     # weekday label
bikes['date'] = bikes['date'].dt.date


# create season_name based on numeric rules
bikes["season"] = np.select(
    [
        (bikes["month"].isin([12, 1, 2])),   # Winter: Dec, Jan, Feb
        (bikes["month"].isin([3, 4, 5])),    # Spring: Mar, Apr, May
        (bikes["month"].isin([6, 7, 8])),    # Summer: Jun, Jul, Aug
        (bikes["month"].isin([9, 10, 11]))   # Autumn: Sep, Oct, Nov
    ],
    ["Winter", "Spring", "Summer", "Autumn"], 
    default="Unknown"
)

# convert to ordered categorical, like fct_relevel in R
season_order = ["Winter", "Spring", "Summer", "Autumn"]
bikes["season"] = pd.Categorical(
    bikes["season"],
    categories=season_order,
    ordered=True
)

bikes.drop('wday', axis=1, inplace=True)

import numpy as np
import pandas as pd
from scipy import stats
import matplotlib.pyplot as plt

# --- Helper to build CIs from summary stats ---
def add_ci(df):
    df = df.copy()
    df['se'] = df['std'] / np.sqrt(df['n'])
    df['tcrit'] = stats.t.ppf(0.975, df['n'] - 1)
    df['lower'] = df['mean'] - df['tcrit'] * df['se']
    df['upper'] = df['mean'] + df['tcrit'] * df['se']
    return df

# ---------- Per season ----------
season_summary = (
    bikes.groupby('season', observed=True)['bikes_hired']
            .agg(mean='mean', n='count', std='std')
            .reset_index()
    .pipe(add_ci)
)

print("Seasonal mean and 95% CI:")
print(season_summary[['season','lower','mean','n','upper']])

# ---------- Per month ----------
month_summary = (
    bikes.groupby(['month','month_name'], observed=True)['bikes_hired']
            .agg(mean='mean', n='count', std='std')
            .reset_index()
    .pipe(add_ci)
    .sort_values('month')
)

print("\nMonthly mean and 95% CI:")
print(month_summary[['month_name','mean','lower','upper','n']])

# ---------- Plots ----------
# Season plot
plt.figure(figsize=(8,5))
yerr = np.vstack([
    season_summary['mean'] - season_summary['lower'],
    season_summary['upper'] - season_summary['mean']
])
plt.errorbar(season_summary['season'], season_summary['mean'],
                yerr=yerr, fmt='o', capsize=5)
plt.title('Mean Bikes Hired per Season with 95% CI')
plt.ylabel('Mean Bikes Hired')
plt.xlabel('Season')
plt.tight_layout()
plt.show()

# Month plot
plt.figure(figsize=(10,5))
yerr_m = np.vstack([
    month_summary['mean'] - month_summary['lower'],
    month_summary['upper'] - month_summary['mean']
])
plt.errorbar(month_summary['month_name'], month_summary['mean'],
                yerr=yerr_m, fmt='o', capsize=5)
plt.title('Mean Bikes Hired per Month with 95% CI')
plt.ylabel('Mean Bikes Hired')
plt.xlabel('Month')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

Interpretation

  1. The true mean demand is very likely to lie within the calculated CI ranges.

  2. Summer has significantly higher demand than winter — the intervals barely overlap (if at all).

  3. Spring and Autumn overlap a lot, meaning their average demands are statistically similar.

  4. Monthly CIs overlap between adjacent months, showing a gradual seasonal transition rather than sharp jumps.

👉 The overlaps tell us whether differences are statistically meaningful. For example, winter is clearly lower than summer, but spring vs autumn is not a clear separation.

import statsmodels.api as sm
import statsmodels.formula.api as smf

# --- 1) Select predictors ---
predictors = [
    "mean_temp", "humidity", "cloud_cover", "precipitation",
    "sunshine", "radiation", "pressure",
    "season", "weekend", "year", "month", "day_of_week"
]

# --- 2) Build formula ---
formula = "bikes_hired ~ " + " + ".join(predictors)

# --- 3) Fit model ---
model = smf.ols(formula=formula, data=bikes).fit()
