# Visualisation: Genre-wise Distribution of IMDB Ratings and Gross Revenue

# Prepare data for plotting
genre_stats = (
    movies.groupby('genre')
    .agg(
        avg_rating=('rating', 'mean'),
        avg_gross=('gross', 'mean'),
        count=('title', 'count')
    )
    .reset_index()
)

# Sort genres by average rating for better visual order
genre_stats = genre_stats.sort_values(by='avg_rating', ascending=False)

fig, ax1 = plt.subplots(figsize=(13, 7))

# Bar plot: Average IMDB rating by genre
sns.barplot(
    data=genre_stats,
    x='genre',
    y='avg_rating',
    hue='genre',
    palette='viridis',
    ax=ax1
)
ax1.set_ylabel('Average IMDB Rating', fontsize=12)
ax1.set_xlabel('Genre', fontsize=12)
ax1.set_xticks(range(len(genre_stats['genre'])))
ax1.set_xticklabels(genre_stats['genre'], rotation=30, ha='right', fontsize=11)
ax1.tick_params(axis='y', labelsize=11)

# Line plot: Average gross revenue by genre (secondary axis)
ax2 = ax1.twinx()
sns.pointplot(
    data=genre_stats,
    x='genre',
    y='avg_gross',
    color='crimson',
    markers='D',
    linestyles='-',
    ax=ax2
)
ax2.set_ylabel('Average Gross Revenue (USD)', fontsize=12, color='crimson')
ax2.tick_params(axis='y', labelcolor='crimson', labelsize=11)
ax2.set_ylim(0, genre_stats['avg_gross'].max() * 1.2)

# Annotate number of movies per genre above bars
for idx, row in genre_stats.iterrows():
    ax1.text(
        idx, row['avg_rating'] + 0.08, f"{int(row['count'])}", 
        ha='center', va='bottom', fontsize=10, color='black'
    )

# Title and layout
plt.title(
    'Genre-wise Average IMDB Rating and Gross Revenue\n(Number above bars = Number of Movies)',
    fontsize=15, pad=20
)
ax1.grid(axis='y', linestyle='--', alpha=0.5)
fig.tight_layout()
plt.show()
